📋 JSON Schema Reference ▾
TOP-LEVEL STRUCTURE
{
  "version": number,      // export format — use 23
  "exportedFrom": string,  // e.g. "1.0.4"
  "minimumVersion": string|null,
  "meta": { /* see below */ },
  "data": { /* see below */ }
}
META
"meta": {
  "name": string,
  "author": string,
  "version": string,      // e.g. "1.0.0"
  "description": string,
  "autoRunAction": null|string
}
DATA
"data": {
  "actions": Action[],
  "queues": [],         // usually empty
  "commands": Command[],
  "timers": Timer[],
  "websocketServers": [],
  "websocketClients": []
}
ACTION
{
  "id": string,           // UUID v4
  "name": string,
  "group": string|null,
  "enabled": boolean,
  "queue": string,       // UUID — use "00000000-0000-0000-0000-000000000000"
  "alwaysRun": boolean,
  "concurrent": boolean,
  "randomAction": boolean,
  "excludeFromHistory": boolean,
  "excludeFromPending": boolean,
  "collapsedGroups": [],
  "triggers": Trigger[],
  "subActions": SubAction[]
}
TRIGGERS (type codes)
// Chat command trigger
{ "type": 401, "commandId": string,
  "id": UUID, "enabled": bool, "exclusions": [] }

// Timer trigger
{ "type": 701, "timerId": string,
  "id": UUID, "enabled": bool, "exclusions": [] }

// Follower / Sub trigger (min/max viewers)
{ "type": 102, "min": number, "max": number,
  "id": UUID, "enabled": bool, "exclusions": [] }

// Stream start (type 124), Stream end (type 133)
{ "type": 124|133,
  "id": UUID, "enabled": bool, "exclusions": [] }

// WebSocket custom event (type 2003)
{ "type": 2003,
  "id": UUID, "enabled": bool, "exclusions": [] }
SUB-ACTIONS (type codes)
// Execute C# code (type 99999)
{
  "type": 99999,
  "id": UUID, "enabled": bool, "index": number,
  "weight": 0, "parentId": null,
  "name": null, "description": null,
  "references": string[],   // DLL paths
  "precompile": false,
  "delayStart": false,
  "saveResultToVariable": false,
  "saveToVariable": null,
  "byteCode": string      // base64(UTF-8 C# source)
}

// Set variable (type 123)
{
  "type": 123, "variableName": string,
  "value": string, "autoType": boolean,
  "id": UUID, "enabled": bool, "index": number,
  "weight": 0, "parentId": null
}

// Delay (type 1002)
{
  "type": 1002, "value": string,  // ms as string
  "maxValue": null, "random": false,
  "id": UUID, "enabled": bool, "index": number,
  "weight": 0, "parentId": null
}

// OBS Source visibility (type 30)
{
  "type": 30, "sceneName": string,
  "sourceName": string, "state": 0|1,
  "connectionId": "00000000-...",
  "id": UUID, "enabled": bool, "index": number,
  "weight": 0, "parentId": null
}

// Send chat message (type 1 or 7)
{
  "type": 1, "message": string,
  "id": UUID, "enabled": bool, "index": number,
  "weight": 0, "parentId": null
}
COMMAND
{
  "id": UUID, "name": string,
  "command": string,        // "!cmd\r\nalias" for multi-alias
  "enabled": boolean,
  "mode": 0,               // 0=contains, 1=exact, 2=startsWith
  "location": 0,
  "sources": 1,
  "globalCooldown": number, // seconds
  "userCooldown": number,
  "grantType": 0,
  "caseSensitive": false,
  "group": null,
  "include": false,
  "ignoreBotAccount": false,
  "ignoreInternal": false,
  "regexExplicitCapture": false,
  "persistCounter": false,
  "persistUserCounter": false,
  "permittedUsers": [],
  "permittedGroups": []
}
TIMER
{
  "id": UUID, "name": string,
  "enabled": boolean,
  "repeat": boolean,
  "interval": number,      // seconds
  "randomInterval": false,
  "upperInterval": 0,
  "lines": 0,
  "counter": 0
}
